#+NAME: /etc/amavis/conf.d/15-av_scanners
#+begin_src
# $ sudo adduser clamav amavis
['ClamAV-clamd',
  \&ask_daemon, ["CONTSCAN {}\n", "/var/run/clamav/clamd.ctl"],
  qr/\bOK$/m,
  qr/\bFOUND$/m,
  qr/^.*?: (?!Infected Archive)(.*) FOUND$/m ],
#+end_src

#+NAME: /etc/clamav/clamav.conf
#+begin_src
LocalSocket /var/run/clamav/clamd.ctl
#+end_src
